\begin{problem}{1.2.4}
    设 $f:\mathbb{D} \to \mathbb{D}$ 是解析函数，其中 $f(0) = 0, a \not\in f(\mathbb{D}),a\in (0,1)$。此时求证
    \begin{equation*}
        |f'(0)| \le \frac{2a\ln(\frac{1}{a})}{1-a^{2}}
    \end{equation*}
    当且仅当
    \begin{equation*}
        f(z) = \frac{a - h(z)}{1- ah(z)},\quad h(z) = e^{\frac{1-e^{\sqrt{-1}\theta}z}{1+e^{\sqrt{-1}\theta}z}\ln a}
    \end{equation*}
    时等号成立。
\end{problem}
